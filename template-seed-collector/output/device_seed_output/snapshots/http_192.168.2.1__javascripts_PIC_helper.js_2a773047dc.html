	function addSetToggleClickHandler(devName,devMAC) {
		var devIDs = ["#" + devName + "_Toggle", "#" + devName + "_BlockedImg", "#" + devName + "_BlockedText", "#" + devName + "_UnblockedImg", "#" + devName + "_UnblockedText"];
		$(devIDs[0]).click(function(e) {
			e.preventDefault();
			if ($(devIDs[1]).is(":hidden")) {
				$(devIDs[1]).show();
				$(devIDs[2]).css('visibility', 'visible');
				$(devIDs[3]).hide();
				$(devIDs[4]).css('visibility', 'hidden');
			} else {
				$(devIDs[1]).hide();
				$(devIDs[2]).css('visibility', 'hidden');
				$(devIDs[3]).show();
				$(devIDs[4]).css('visibility', 'visible');
			}
			$.post("/goform/formParConSave", {
				work:"2",
				mac:devMAC,
				block:($("#"+devName+"_BlockedImg").is(":hidden")?0:1),
			}, function(data){});
		});
	}

	function addSaveHandler(devName,devMAC) {
		var devIDs = ["#" + devName + "_save", "#" + devName + "_details", "#" + devName + "Name", "#" + devName + "NameInput", "#" + devName + "_caret", "#" + devName + "_DetailsToggle", "#" + devName + "_msgSaved"];
		$(devIDs[0]).click(function(e) {
			if($(devIDs[0]).hasClass("disabled"))
				return
			if($("#"+devName+"_SNFrom").get(0).selectedIndex > $("#"+devName+"_SNTo").get(0).selectedIndex){
				$("#"+devName+"_SNTo").get(0).selectedIndex = $("#"+devName+"_SNFrom").get(0).selectedIndex
			}	
			if($("#"+devName+"_WEFrom").get(0).selectedIndex > $("#"+devName+"_WETo").get(0).selectedIndex){
				$("#"+devName+"_WETo").get(0).selectedIndex = $("#"+devName+"_WEFrom").get(0).selectedIndex
			}	
			if ($(devIDs[3]).val() == "")
				$(devIDs[3]).val("Unnamed Device")
			$(devIDs[2]).html($(devIDs[3]).val().replace(/ /g,"&nbsp;"))
			$("#" + devName + "_footer").html($(devIDs[3]).val().replace(/ /g,"&nbsp;")+" "+langDictionary['gtSoloSaved'])
			$("#"+devName+"_ToggleSched").is(":checked")?$("#" + devName + "_clockIcon").css({"visibility":"visible","display":"block"}):$("#" + devName + "_clockIcon").css({"visibility":"hidden","display":"none"})
			
			$(devIDs[2]).show();
			$(devIDs[3]).hide();
			
			$(devIDs[4]).rotate({animateTo:180, duration:600});

			// Show a little message to say changes were saved
			$(devIDs[1]).slideUp("slow", function () {
			  $(devIDs[6]).slideDown("slow", function () {
				$(devIDs[6]).animate({opacity: 1}, 3000, function () {
					$(devIDs[6]).slideUp("slow");
				});
			  });
			});
			
			// disable the button until something else changes
			$(devIDs[0]).addClass("disabled");
			//if ($("#"+devName+"NameInput").val() == "")
			//	$("#"+devName+"NameInput").val("Unnamed Device")
			$.post("/goform/formParConSave", {
				work:"1",
				mac:devMAC,
				name:$("#"+devName+"NameInput").val().replace(/[ "]/g,"_"),
				schedule:($("#"+devName+"_ToggleSched").is(":checked")?1:0),
				from:$("#"+devName+"_SNFrom").get(0).selectedIndex,
				to:(($("#"+devName+"_SNTo").get(0).selectedIndex)+1)%24,
				from1:$("#"+devName+"_WEFrom").get(0).selectedIndex,
				to1:(($("#"+devName+"_WETo").get(0).selectedIndex)+1)%24
			}, function(data){});
		});
	}
	
	function addFormChangeHandler(devName) {
		var devIDs = ["#" + devName + "_save", "#" + devName + "_name", "#" + devName + "NameInput", "#" + devName + "_SNFrom", "#" + devName + "_SNTo", "#" + devName + "_WEFrom", "#" + devName + "_WETo"];
		$(devIDs[2]).change(function(e) {
			// Enable the save button when something changes.
			$(devIDs[0]).removeClass("disabled");
		});
		$(devIDs[3]).change(function(e) {
			$(devIDs[4]).get(0).selectedIndex = $(devIDs[3]).get(0).selectedIndex
			// Enable the save button when something changes.
			$(devIDs[0]).removeClass("disabled");
		});
		$(devIDs[5]).change(function(e) {
			$(devIDs[6]).get(0).selectedIndex = $(devIDs[5]).get(0).selectedIndex
			// Enable the save button when something changes.
			$(devIDs[0]).removeClass("disabled");
		});
		$(devIDs[4]).change(function(e) {
			// Enable the save button when something changes.
			$(devIDs[0]).removeClass("disabled");
		});
		$(devIDs[6]).change(function(e) {
			// Enable the save button when something changes.
			$(devIDs[0]).removeClass("disabled");
		});
	}
	
	function addSetToggleSchedHandler(devName) {
		var devIDs = ["#" + devName + "_ToggleSched", "#" + devName + "_details>table", "#" + devName + "_details>input", "#" + devName + "_save"];
		$(devIDs[0]).click(function(e) {
			if (this.checked) {
				$(devIDs[1]).removeClass("PICInactive");
			} else {
				$(devIDs[1]).addClass("PICInactive");
			}
			// Enable the save button when something changes.
			$(devIDs[3]).removeClass("disabled");
		});
	}
	
	function addShowDetailsHandler(devName) {
		var devIDs = ["#" + devName + "_DetailsShow", "#" + devName + "_details", "#" + devName + "Name", "#" + devName + "NameInput", "#" + devName + "_caret", "#" + devName + "_DetailsToggle"];
		$(devIDs[0]).click(function(e) {
			e.preventDefault();
			if ($(devIDs[1]).is(":hidden")) {
				$(devIDs[2]).hide();
				$(devIDs[3]).show();
				$(devIDs[1]).slideDown();
				
				$(devIDs[4]).rotate({animateTo:0, duration:600});
			}
		});
		$(devIDs[5]).click(function(e) {
			e.preventDefault();
			if ($(devIDs[1]).is(":hidden")) {
				$(devIDs[2]).hide();
				$(devIDs[3]).show();
				$(devIDs[1]).slideDown();
				
				$(devIDs[4]).rotate({animateTo:0, duration:600});
			} else {
				$(devIDs[2]).show();
				$(devIDs[3]).hide();
				$(devIDs[1]).slideUp();
				
				$(devIDs[4]).rotate({animateTo:180, duration:600});
			}
		});
	}
