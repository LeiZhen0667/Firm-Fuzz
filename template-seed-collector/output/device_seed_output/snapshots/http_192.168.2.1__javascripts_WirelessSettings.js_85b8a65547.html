function isNOnly(modeVal) {
	if (modeVal === '2' || modeVal === 'ac' || modeVal === 'nac') {
		return true;
	}
	return false;
}

function includesN(modeVal) {
	//if (modeVal.indexOf('n') !== -1 || modeVal === 'ac') {
	if (modeVal == 2 | modeVal == 3 | modeVal == 4 ) {
		return true;
	}
	return false;
}

	// We're not selling any G only routers!
	//MaxBandwidth24: 40,	

	if (caps.MaxBandwidth5 == 80) {

function displaySettings(idPrefix, net) {
	$(idPrefix + "SSID").val(net.SSID);
	$(idPrefix +"PSK").val(net.Passkey);
	if ($(idPrefix + "Security option[value='" + net.Auth + "']").length > 0) {
		$(idPrefix + "Security option[value='" + net.Auth + "']").prop('selected', true);
	} else {
		// For the WEP/Open case
		$(idPrefix +"Security option[value='" + net.Encryption + "'").prop('selected', true);
	}
	// Store the previous value
	$(idPrefix + "Security").data("previous-value", $(idPrefix + "Security").val());

	// Show the correct status if the network is not Enabled
	if (!net.Enabled) {
		$(idPrefix + "EnableOff").show();
		$(idPrefix + "EnableOn").hide();
	}
	$(idPrefix + "EnableVal").val(net.Enabled == true);
	$(idPrefix + "Chan option[value=\"" + net.Channel + "\"]").prop('selected', true);
	if (net.Channel == "constAuto" || net.Bandwidth == 20) {
		$(idPrefix + "Ext").prop('disabled', true);
	} else {
		$(idPrefix + "Ext").prop('disabled', false);
	}
	$(idPrefix + "Broadcast").attr('checked', net.BroadcastSSIDEnabled?true:false);
	
	$(idPrefix + "Ext option[value=\"" + net.ExtensionChannel + "\"]").prop('selected', true);
	$(idPrefix + "BW option[value=\"" + net.Bandwidth + "\"]").prop('selected', true);
	$(idPrefix + "Mode option[value=\"" + net.WirelessMode + "\"]").prop('selected', true);
	$(idPrefix + "Mode").data("previous-value", $(idPrefix +"Mode").val());
	if (includesN(net.WirelessMode)) {
		$(idPrefix + "ModeWarning").hide();	
	} else {
		$(idPrefix + "ModeWarning").slideDown();
	}
	setBW(idPrefix + "Mode", net.WirelessMode);
}

/// Event handlers
function setPSKVisible(secID) {
	var newMode;
	var modeID = secID.indexOf("24")>=0?"#p2500f24Mode":"#p2500f5Mode";
	var PSKid = secID.indexOf("24")>=0?"#p2500f24PSK":"#p2500f5PSK";
	var errPSKid = secID.indexOf("24")>=0?"#p2500e24PSK":"#p2500e5PSK";
	if ($("#"+secID).val() == "constNone") {
		$(PSKid).css("visibility", "hidden");
		$(errPSKid).hide();
	} else {
		$(PSKid).css("visibility", "visible");
		validatePSK_single(PSKid.substring(0,PSKid.length-3), errPSKid.substring(0, errPSKid.length-3)); 
	}
	
	//console.log("Updating to " + secID + ", new Value: " + $("#"+secID).val() + " prev Value:" + $("#"+secID).data("previous-value"));
	// Check if we need to update the mode because we went to/from WEP
	if ($("#"+secID).val() == "constWEP") {
		// We changed TO WEP
		if (includesN($(modeID).val())) {
			// N modes are not allowed in WEP.  Auto-select most permissive mode
			var newMode = secID.indexOf("24")>=0?"5":"1";

			$(modeID + " option[value=\"" + newMode + "\"]").prop('selected', true);
			// Trigger any updates to the rest of the Adv Options
			setBW(modeID, newMode);
			$(modeID + "Warning").slideDown();
		}
	} else if ($("#"+secID).val() == "constWPA2Personal") {
		// non-N modes are not allowed in WPA2.  Auto-select most permissive mode
		var newMode = "2";
		if ($(modeID +" option[value='ac']").length > 0)
			newMode = "nac";

		$(modeID + " option[value=\"" + newMode + "\"]").prop('selected', true);
		// Trigger any updates to the rest of the Adv Options
		setBW(modeID, newMode);
		$(modeID + "Warning").slideUp();
	} else if ($("#"+secID).val() == "constWPAPersonalMixed" && isNOnly($(modeID).val())) {
		// non-N modes are needed for mixed mode.  Auto-select most permissive mode
		//var newMode = secID.indexOf("24")>=0?"bgn":"an";
		var newMode = "3";
		if ($(modeID +" option[value='ac']").length > 0)
			newMode = "anac";
			
		$(modeID + " option[value=\"" + newMode + "\"]").prop('selected', true);
		// Trigger any updates to the rest of the Adv Options
		setBW(modeID, newMode);
		$(modeID + "Warning").slideUp();
	} else if ($("#"+secID).val() != "constWEP" && 
		!includesN($(modeID).val())) {
		// We changed AWAY from WEP
		// N modes are now allowed in non-WEP.  Auto-select most permissive mode
		//var newMode = secID.indexOf("24")>=0?"bgn":"an";
		var newMode = "3";
		if ($(modeID +" option[value='ac']").length > 0)
			newMode = "anac";

		$(modeID + " option[value=\"" + newMode + "\"]").prop('selected', true);
		// Trigger any updates to the rest of the Adv Options
		setBW(modeID, newMode);
		$(modeID + "Warning").slideUp();
	}
	$(modeID).data("previous-value", $(modeID).val());
}

function copyVal() {
	// Save the value entered
	//origCheckbox[this.id] = $(this).val();
	if (!$('#p2500fSame').attr('checked'))
		return;
	
	// Copy the value
	var idOther = this.id.indexOf("24")>=0?this.id.replace("f24", "f5"):this.id.replace("f5", "f24");
	$("#"+idOther).val($(this).val());
	
	// Update the PSK, in case it changed modes.
	var secID = "p2500f5Security", secIDOther = "p2500f24Security";
	if (this.id.indexOf("24")>=0) {
		secID = "p2500f24Security";
		secIDOther = "p2500f5Security";		
	}
	setPSKVisible(secID);
	setPSKVisible(secIDOther);
}

function validateSSID() {

function validatePSK_single(idPrefix, errPrefix) {
	var sec = $(idPrefix + 'Security').val();
	if (sec != "constNone") {
		var psk = $(idPrefix + 'PSK').val();
		if (psk == "") {
			$(errPrefix + 'PSK').text(langDictionary["errorWPAPasswordEmpty"]).fadeTo(200, 1);
		} else if (sec == "constWEP") {
			if (!(psk.length == 10 || psk.length == 26)) {
				$(errPrefix + 'PSK').text(langDictionary["errorWEPPasswordInvalidLength"]).fadeTo(200, 1);
			} else if (!psk.match("^[a-fA-F0-9_]*$")) {
				$(errPrefix + 'PSK').text(langDictionary["errorWEPPasswordInvalidChars"]).fadeTo(200, 1);
			} else {
				$(errPrefix + 'PSK').hide();
			}
		} else {
			if (psk.length < 8) {
				$(errPrefix + 'PSK').text(langDictionary["errorWPAPasswordInvalidLength"]).fadeTo(200, 1);
			} else if( validStr(psk) == false || validVisibleStr(psk) == false){
				$(errPrefix + 'PSK').text(langDictionary["errorWPAPasswordInvalidChars"]).fadeTo(200, 1);
			} else {
				$(errPrefix + 'PSK').hide();
			}
		}
	} else {
		$(errPrefix + 'PSK').hide();
	}
}

function validatePSK() {
	validatePSK_single("#p2500f24", "#p2500e24");
	validatePSK_single("#p2500f5", "#p2500e5");
}

function readNetwork(idPrefix) {
	var sec = $(idPrefix + 'Security').val();
	var netObj = {
		"Passkey": $(idPrefix + 'PSK').val(), 
		"SSID": $(idPrefix + 'SSID').val(), 
		"Auth": sec, 
		"Encryption": sec,
		"Channel": $(idPrefix + "Chan option").val(),
		"ExtensionChannel": $(idPrefix + "Ext option").val(),
		"Bandwidth": $(idPrefix + "BW option").val(),
		"WirelessMode" : $(idPrefix + "Mode option").val(),
		"Enabled" : ($(idPrefix + "EnableVal").val() == "true")
	}
	if (sec == "constWPA2Personal")
		netObj.Encryption = "constAES";
	if (sec == "constWPAPersonalMixed")
		netObj.Encryption = "constAES";
	if (sec == "constWEP")
		netObj.Auth = "constNone";
	
	return netObj;
}

function waitForRouter() {
console.log("waitForRouter");

		type: "GET",
		url: "../BelkinAPI/IsClientWireless",
		data: { RequestID: Math.floor(Math.random() * 10000) },
